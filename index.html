let audioCtx; function initAudio() { if (!audioCtx) { audioCtx = new (window.AudioContext || window.webkitAudioContext)(); } } function playBeep(freq, duration) { const osc = audioCtx.createOscillator(); const gain = audioCtx.createGain(); osc.connect(gain); gain.connect(audioCtx.destination); osc.frequency.value = freq; osc.type = "sine"; osc.start(); osc.stop(audioCtx.currentTime + duration); } // 🎨 ИСПРАВЛЕННАЯ смена цвета function setColor(color) { document.body.style.color = color; document.querySelectorAll("button").forEach(btn => { btn.style.borderColor = color; btn.style.color = color; btn.style.boxShadow = `0 0 10px ${color}`; }); document.getElementById("timer").style.textShadow = `0 0 10px ${color}, 0 0 20px ${color}, 0 0 40px ${color}`; } // 🌍 переводы const translations = { ru: { round:"Раунд", rest:"Отдых", end:"Бой окончен!" }, en: { round:"Round", rest:"Rest", end:"Fight over!" }, uk: { round:"Раунд", rest:"Відпочинок", end:"Бій закінчено!" }, cs: { round:"Kolo", rest:"Odpočinek", end:"Konec zápasu!" }, pl: { round:"Runda", rest:"Przerwa", end:"Koniec walki!" } }; // ⏱ настройки let roundTime = 180; let restTime = 60; let totalRounds = 5; let currentRound = 1; let isRest = false; let seconds = roundTime; let interval = null; // элементы const timerEl = document.getElementById("timer"); const roundEl = document.getElementById("round"); const langSelect = document.getElementById("langSelect"); // обновление function update() { let m = Math.floor(seconds / 60); let s = seconds % 60; timerEl.textContent = String(m).padStart(2,'0') + ":" + String(s).padStart(2,'0'); const t = translations[langSelect.value]; roundEl.textContent = isRest ? `${t.rest} ${currentRound}` : `${t.round} ${currentRound}`; } // таймер function tick() { if (seconds <= 10 && seconds > 0) { playBeep(800, 0.1); } if (seconds > 0) { seconds--; } else { playBeep(200, 0.5); if (!isRest) { isRest = true; seconds = restTime; } else { isRest = false; currentRound++; if (currentRound > totalRounds) { clearInterval(interval); timerEl.textContent = "00:00"; roundEl.textContent = translations[langSelect.value].end; return; } seconds = roundTime; } playBeep(200, 0.5); } update(); } // кнопки document.getElementById("start").onclick = () => { initAudio(); // 🔥 ВАЖНО audioCtx.resume(); if (!interval) interval = setInterval(tick, 1000); }; document.getElementById("stop").onclick = () => { clearInterval(interval); interval = null; }; document.getElementById("reset").onclick = () => { clearInterval(interval); interval = null; currentRound = 1; isRest = false; seconds = roundTime; update(); }; langSelect.onchange = update; update();
